; Desde $C000 hasta $CFFF y
; desde $D800 hasta $FFFF

	PLA
	LDA #0
	STA 203
	LDA #192
	STA 204
	SEI
	LDA #0
	STA 54286
	LDY #0
COPY:
	LDA #253
	STA 54017
	LDA (203),Y
	DEC 54017
	STA (203),Y
	INY
	BNE COPY
	INC 204
	LDA 204
	CMP 208
	BNE NOBLOCK2
	LDA #216
	STA 204

	BNE COPY

	STA 54286
	CLI
